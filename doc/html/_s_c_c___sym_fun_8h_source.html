992  int create(const char** V, int Vcount, const char** C,
993  int Ccount, double const* Cvalues, char const* S)
994  {
995  //
996  // Allocate Storage and Initalize
997  //
998  constructorString = new char[strlen(S) + 1];
999  COPYSTR(constructorString, strlen(S) + 1,S);
1000 
1001  variableNames = new char*[Vcount];
1002  variableCount = Vcount;
1003 
1004  int i;
1005  for(i = 0; i < Vcount; i++)
1006  {
1007  variableNames[i] = new char[strlen(V[i]) + 1];
1008  COPYSTR(variableNames[i], strlen(V[i]) + 1,V[i]);
1009  }
1010 
1011  constantNames = new char*[Ccount];
1012  constantCount = Ccount;
1013  constantValues = new double[Ccount];
1014 
1015  for(i = 0; i < Ccount; i++)
1016  {
1017  constantNames[i] = new char[strlen(C[i]) + 1];
1018  COPYSTR(constantNames[i], strlen(C[i]) + 1,C[i]);
1019  constantValues[i] = Cvalues[i];
1020  }
1021 
1022  RealOperatorLib L;
1023  ExpressionTransform T;
1024  int expReturn;
1025 
1026  expReturn = T.initialize(variableNames, variableCount, constantNames,
1027  constantCount,constructorString,&L);
1028  if(expReturn != 0) {destroy(); return 1;}
1029 
1030  evaluationDataSize = T.getEvaluationDataSize();
1031  evaluationData = new double[evaluationDataSize];
1032 
1033  executionArraySize = T.getExecutionArraySize();
1034  executionArray = new long[executionArraySize];
1035 
1036  initializeEvaluationData(T);
1037  initializeExecutionArray(T);
1038  //
1039  // Save symbols
1040  //
1041  symbolCount = T.getSymbolCount();
1042  char** TsNames = T.getSymbolNamesPtr();
1043  sNames = new char*[symbolCount];
1044  for(i=0; i< symbolCount; i++)
1045  {
1046  sNames[i] = new char[strlen(TsNames[i])+1];
1047  COPYSTR(sNames[i], strlen(TsNames[i]) + 1,TsNames[i]);
1048  }
1049 
1050  LibFunctions = RealOpLib.FunctionArray;
1051  return 0;
1052  }
