1021  int create(const char** V, int Vcount, const char** C,
1022  int Ccount, double const* Cvalues, char const* S)
1023  {
1024  //
1025  // Allocate Storage and Initalize
1026  //
1027 
1028  constructorString = new char[strlen(S) + 1];
1029  COPYSTR(constructorString, strlen(S) + 1,S);
1030 
1031  variableCount = Vcount;
1032  if(Vcount > 0)
1033  {
1034  variableNames = new char*[Vcount];
1035  }
1036 
1037 
1038  int i;
1039  for(i = 0; i < Vcount; i++)
1040  {
1041  variableNames[i] = new char[strlen(V[i]) + 1];
1042  COPYSTR(variableNames[i], strlen(V[i]) + 1,V[i]);
1043  }
1044 
1045  constantCount = Ccount;
1046 
1047  if(Ccount > 0)
1048  {
1049  constantNames = new char*[Ccount];
1050  constantValues = new double[Ccount];
1051  }
1052 
1053  for(i = 0; i < Ccount; i++)
1054  {
1055  constantNames[i] = new char[strlen(C[i]) + 1];
1056  COPYSTR(constantNames[i], strlen(C[i]) + 1,C[i]);
1057  constantValues[i] = Cvalues[i];
1058  }
1059 
1060  RealOperatorLib L;
1061  ExpressionTransform T;
1062 
1063  int expReturn;
1064 
1065  expReturn = T.initialize(variableNames, variableCount, constantNames,
1066  constantCount,constructorString,&L);
1067  if(expReturn != 0) {destroy(); return 1;}
1068 
1069  evaluationDataSize = T.getEvaluationDataSize();
1070  evaluationData = new double[evaluationDataSize];
1071 
1072  executionArraySize = T.getExecutionArraySize();
1073  executionArray = new long[executionArraySize];
1074 
1075  initializeEvaluationData(T);
1076  initializeExecutionArray(T);
1077  //
1078  // Save symbols
1079  //
1080  symbolCount = T.getSymbolCount();
1081  char** TsNames = T.getSymbolNamesPtr();
1082 
1083  if(symbolCount > 0)
1084  {
1085  sNames = new char*[symbolCount];
1086  }
1087 
1088  for(i=0; i< symbolCount; i++)
1089  {
1090  sNames[i] = new char[strlen(TsNames[i])+1];
1091  COPYSTR(sNames[i], strlen(TsNames[i]) + 1,TsNames[i]);
1092  }
1093 
1094  LibFunctions = RealOpLib.FunctionArray;
1095  return 0;
1096  }
